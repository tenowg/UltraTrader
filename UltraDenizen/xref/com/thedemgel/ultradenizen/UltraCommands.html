18      @Override
19      public void execute(ScriptEntry scriptEntry) throws CommandExecutionException {
20          Action action = (Action) scriptEntry.getObject("action");
21          dShop shop = (dShop) scriptEntry.getObject("shop");
22          String itemid = scriptEntry.hasObject("itemid") ? (String) scriptEntry.getObject("itemid") : null;
23          double price = scriptEntry.hasObject("price") ? (double) scriptEntry.getObject("price") : 0;
24  
25          switch (action) {
26              case SETSELLPRICE:
27                  if (!shop.getShop().getPriceList().containsKey(itemid)) {
28                      dB.echoError("Shop does not contain item " + itemid);
29                      return;
30                  }
31                  shop.getShop().getPriceList().get(itemid).setSellPrice(BigDecimal.valueOf(price));
32                  break;
33              case SETBUYPRICE:
34                  if (!shop.getShop().getPriceList().containsKey(itemid)) {
35                      dB.echoError("Shop does not contain item " + itemid);
36                      return;
37                  }
38                  shop.getShop().getPriceList().get(itemid).setBuyPrice(BigDecimal.valueOf(price));
39                  break;
40          }
41      }

45          // Iterate through arguments
46          for (aH.Argument arg : aH.interpret(scriptEntry.getArguments())) {
47              if (!scriptEntry.hasObject("action") && arg.matchesEnum(Action.values())) {
48                  scriptEntry.addObject("action", Action.valueOf(arg.getValue()));
49              } else if (!scriptEntry.hasObject("shop") && arg.matchesArgumentType(dShop.class)) {
50                  scriptEntry.addObject("shop", dShop.valueOf(arg.toString()));
51              } else if (!scriptEntry.hasObject("itemid") && arg.getPrefix().matches("item")) {
52                  scriptEntry.addObject("itemid", arg.getValue());
53              } else if (!scriptEntry.hasObject("price") && arg.matchesPrimitive(aH.PrimitiveType.Double)) {
54                  scriptEntry.addObject("price", Double.valueOf(arg.getValue()));

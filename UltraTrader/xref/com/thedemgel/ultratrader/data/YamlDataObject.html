307 		public SaveShop(Shop shop) {

312 		@Override
313 		public void run() {
314             try {
315                 ConfigurationSection catSection = config.getConfig().createSection("category");
316 
317                 for (CategoryItem categoryItem : shop.getCategoryItem().values()) {
318                     catSection.set(categoryItem.getCategoryId() + ".id", categoryItem.getCategoryId());
319                     catSection.set(categoryItem.getCategoryId() + ".itemStack", categoryItem);
320                     catSection.set(categoryItem.getCategoryId() + ".slot", categoryItem.getSlot());
321                     catSection.set(categoryItem.getCategoryId() + ".lore", categoryItem.getLore());
322                 }
323 
324                 ConfigurationSection priceConfig = config.getConfig().createSection("priceList");
325 
326                 for (ItemPrice priceList : shop.getPriceList().values()) {
327                     priceConfig.set(priceList.getId() + ".description", priceList.getDescription());
328                     priceConfig.set(priceList.getId() + ".category", priceList.getCategoryId());
329                     priceConfig.set(priceList.getId() + ".itemStack", priceList.getItemStack());
330                     priceConfig.set(priceList.getId() + ".sellPrice", priceList.getSellPrice());
331                     priceConfig.set(priceList.getId() + ".buyPrice", priceList.getBuyPrice());
332                     priceConfig.set(priceList.getId() + ".random", priceList.getId());
333                     priceConfig.set(priceList.getId() + ".slot", priceList.getSlot());
334                 }
335 
336 			    ConfigurationSection invConfig = config.getConfig().getConfigurationSection("inventory");
337 			    if (invConfig == null) {
338 				    invConfig = config.getConfig().createSection("inventory");
339 			    }
340 
341 			    // Save Inventory INFO
342 			    ConfigurationSection inventoryInfoConfig = invConfig.getConfigurationSection("info");
343 			    if (inventoryInfoConfig == null) {
344 				    inventoryInfoConfig = invConfig.createSection("info");
345 				    ConfigValue<String> defaultinventory = new ConfigValue(InventoryInterfaceHandler.DEFAULT_INVENTORY_TYPE);
346 				    inventoryInfoConfig.set("type", defaultinventory.getValue());
347 			    }
348 
349 			    for (Entry<String, ConfigValue> inventoryinfo : shop.getInventoryinfo().entrySet()) {
350 				    inventoryInfoConfig.set(inventoryinfo.getKey(), inventoryinfo.getValue().getValue());
351 			    }
352 
353 			    // Save Inventory items
354 			    ConfigurationSection invconf = invConfig.createSection("items");
355 
356 			    Integer count = 0;
357 
358 			    for (Map.Entry<ItemStack, Integer> entry : shop.getInventory().entrySet()) {
359 				    invconf.set(count.toString() + ".itemstack", entry.getKey());
360 				    invconf.set(count.toString() + ".amount", entry.getValue());
361 				    count++;
362 			    }
363 
364 			    // Get wallet information
365 			    ConfigurationSection walletconfig = config.getConfig().getConfigurationSection("wallet");
366 			    if (walletconfig == null) {
367 				    walletconfig = config.getConfig().createSection("wallet");
368 				    walletconfig.set("type", WalletHandler.DEFAULT_WALLET_TYPE);
369 			    }
370 
371 			    for (Entry<String, ConfigValue> walletinfo : shop.getWalletinfo().entrySet()) {
372 				    walletconfig.set(walletinfo.getKey(), walletinfo.getValue().getValue());
373 			    }
374 
375 			    ConfigurationSection infoconfig = config.getConfig().getConfigurationSection("info");
376 			    if (infoconfig == null) {
377 				    infoconfig = config.getConfig().createSection("info");
378 				    //ConfigValue<String> defaultname = new ConfigValue(L.getString("general.newshopname"));
379 				    infoconfig.set("name", new ConfigValue(L.getString("general.newshopname")));
380 			    }
381 
382 			    for (Entry<String, ConfigValue> info : shop.getInfo().entrySet()) {
383 				    infoconfig.set(info.getKey(), info.getValue().getValue());
384 			    }
385 
386 			    ConfigurationSection blockConfig = config.getConfig().getConfigurationSection("blocks");
387                 if (blockConfig == null) {
388                     blockConfig = config.getConfig().createSection("blocks");
389                 }
390 
391                 ConfigurationSection blocks = blockConfig.createSection("blocklocation");
392 
393                 int i = 0;
394                 for (Location loc : shop.getBlockShops()) {
395                     blocks.set(i + ".world", loc.getWorld().getName());
396                     blocks.set(i + ".x", loc.getBlockX());
397                     blocks.set(i + ".y", loc.getBlockY());
398                     blocks.set(i + ".z", loc.getBlockZ());
399                     i++;
400                 }
401 
402 			    config.saveConfig();
403             } catch (Exception e) {
404                 System.out.println("Store Config cannot be saved (" + e.getMessage() + ")");
405             }

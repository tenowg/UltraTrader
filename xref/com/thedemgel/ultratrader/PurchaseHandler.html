14  /**
15   * If there is anything left (no order is required) a reverse order will be
16   * done, and price will be reduced by anything that remains.
17   *
18   * Order for secure transactions:
19   *
20   * 1) figure price for all items in "Buy Box" - "Buy Stack" 2) Check if player
21   * has available funds 3) Withdraw funds - hold funds in temp variable * 4)
22   * Remove items (also check for availability) from traders inventory * 5) If
23   * difference, remove from players "Buy Box" 6) Attempt to put all items in
24   * players inventory 7) Any remaining items not put into inventory - figure
25   * price value 8) subtract remaining value from holding 9) refund funds to
26   * player account 10) deposit funds into trader account 11) return any remaining
27   * items to trader inventory.
28   */

38  	public static void processPurchase(Shop shop, Player player, ItemStack buyStack) {
39  		boolean charge = true;
40  		// Should we charge
41  		if (shop.isOwner(player)) {
42  			charge = false;
43  		}
44  
45  		String buyStackId = shop.getItemId(buyStack);
46  		ItemPrice invItem = shop.getSellprices().get(buyStackId);
47  		BigDecimal buyStackPriceEach = invItem.getPrice();
48  
49  		BigDecimal buyStackPrice = buyStackPriceEach.multiply(BigDecimal.valueOf(buyStack.getAmount()));
50  
51  		EconomyResponse heldFunds;
52  		if (!charge) {
53  			heldFunds = new EconomyResponse(0, 0, ResponseType.SUCCESS, "");
54  		} else if (UltraTrader.getEconomy().has(player.getName(), player.getWorld().getName(), buyStackPrice.doubleValue())) {
55  			heldFunds = UltraTrader.getEconomy().withdrawPlayer(player.getName(), player.getWorld().getName(), buyStackPrice.doubleValue());
56  		} else {
57  			player.sendMessage(L.getString("transaction.sale.player.notenoughfunds"));
58  			return;
59  		}
60  
61  		if (heldFunds == null) {
62  			player.sendMessage(L.getString("transaction.error.economy"));
63  			return;
64  		}
65  
66  		if (!heldFunds.type.equals(ResponseType.SUCCESS)) {
67  			player.sendMessage(L.getString("transaction.error.economy"));
68  			return;
69  		}
70  
71  		// Check for availability
72  		ItemStack baseItem = invItem.getItemStack();
73  		Integer currentInvAmount = shop.getInventoryInterface().getInventoryStock(baseItem);
74  
75  		if (buyStack.getAmount() > currentInvAmount) {
76  			if (charge) {
77  				UltraTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), heldFunds.amount);
78  			}
79  			// Some error checks for deposit
80  			player.sendMessage(L.getString("transaction.sale.shop.notenoughitems"));
81  			return;
82  		} else {
83  			shop.getInventoryInterface().removeInventory(baseItem, buyStack.getAmount());
84  		}
85  
86  		ItemStack placedStack = invItem.getItemStack().clone();
87  		placedStack.setAmount(buyStack.getAmount());
88  		HashMap<Integer, ItemStack> returnedItems = player.getInventory().addItem(placedStack);
89  
90  		BigDecimal refund = BigDecimal.ZERO;
91  		Integer finalamount = buyStack.getAmount();
92  
93  		if (!returnedItems.isEmpty()) {
94  			// calculate returned funds... readd to merchant
95  			Integer returned = 0;
96  			for (ItemStack returnedItem : returnedItems.values()) {
97  				returned = returned + returnedItem.getAmount();
98  			}
99  
100 			if (charge) {
101 				refund = buyStackPriceEach.multiply(BigDecimal.valueOf(returned));
102 
103 				UltraTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), refund.doubleValue());
104 			}
105 
106 			ItemStack itemReturned = baseItem.clone();
107 			itemReturned.setAmount(returned);
108 			shop.getInventoryInterface().addInventory(itemReturned);
109 			finalamount =- returned;
110 		}
111 
112 		BigDecimal traderDeposit = BigDecimal.valueOf(heldFunds.amount).subtract(refund);
113 
114 		// Deposit into trader when interface is ready.
115 		if (!charge) {
116 			player.sendMessage(L.getFormatString("transaction.sale.shop.purchase", buyStack.getType().name(), finalamount));
117 			player.sendMessage(L.getFormatString("transaction.sale.shop.totalpurchase", "FREE"));
118 		} else if (shop.getWallet().addFunds(traderDeposit).type.equals(ResponseType.SUCCESS)) {
119 			player.sendMessage(L.getFormatString("transaction.sale.shop.purchase", buyStack.getType().name(), finalamount));
120 			player.sendMessage(L.getFormatString("transaction.sale.shop.totalpurchase", UltraTrader.getEconomy().format(traderDeposit.doubleValue())));
121 		} else {
122 			player.sendMessage(L.getString("transaction.error.fundstoshop"));
123 		}
124 
125 		shop.save();
126 	}

83          current = Status.CATEGORY_SCREEN;
84  
85          top.clear();

87          boolean displayAdmin = shop.getInventoryInterface().displayItemToPlayer(player);
88  
89          List<CategoryItem> itemQueue = new ArrayList<>();
90  
91          for (CategoryItem item : shop.getCategoryItem().values()) {
92              if (item.getSlot() == -1) {
93                  itemQueue.add(item);
94              } else {
95                  ItemStack slot = getTopInventory().getItem(item.getSlot());
96  
97                  if (slot != null) {
98                      itemQueue.add(item);
99                  } else {

103         }
104 
105         for (CategoryItem item : itemQueue) {
106             int slot = getTopInventory().firstEmpty();
107             item.setSlot(slot);
108             this.getTopInventory().setItem(item.getSlot(), item);
109         }
110 
111         // Check if limits allow for remote access...
112         if (shop.getCanRemote()) {
113             ItemStack doItemShop = new ItemStack(StoreItem.STORE_ITEM_MATERIAL);
114             StoreItem.linkToShop(shop, doItemShop);

260 		}
261 
262 		int invCount = shop.getInventoryInterface().getInventoryStock(invItem);
263 
264         Map<Integer, ? extends ItemStack> itemStackHashMap = player.getInventory().all(invItem.getItemStack().getType());
265 
266         int max = 0;
267 
268         for (ItemStack stack : itemStackHashMap.values()) {
269             if (stack.isSimilar(invItem.getItemStack())) {
270                 max += stack.getAmount();
271             }
272         }
273 
274 		int count = 1;
275 
276 		while (count <= max && count <= 64) {
277             top.addItem(invItem.generateLore(count, current));

429     public void setCategory(final int slot) {
430         Predicate<CategoryItem> categoryItemPredicate = new Predicate<CategoryItem>() {
431             @Override
432             public boolean apply(@Nullable CategoryItem categoryItem) {
433                 return categoryItem.getSlot() == slot;
434             }
435         };
436 
437         Collection<CategoryItem> categoryItems = Collections2.filter(shop.getCategoryItem().values(), categoryItemPredicate);
438 
439         if (categoryItems.size() > 0) {
440             category = categoryItems.iterator().next().getCategoryId();
441         }
442     }

57  	public void onInventoryClickEvent(final InventoryClickEvent event) {
58  		Player player = (Player) event.getWhoClicked();
59  
60  		if (!(event.getView() instanceof ShopInventoryView)) {
61  			return;
62  		}
63  		// Clear out event types we don't want to handle
64  		if (event.getAction().equals(InventoryAction.MOVE_TO_OTHER_INVENTORY)) {
65  			event.setCancelled(true);
66  			return;
67  		}
68  
69  		final ShopInventoryView view = (ShopInventoryView) event.getView();
70  
71  		// Nothing needs to be dragged anymore...
72  		if (!(event.getRawSlot() < view.getTopInventory().getSize())) {
73  			return;
74  		} else {
75  			event.setCancelled(true);
76  		}
77  
78  		if (view.getStatus().equals(Status.MAIN_SCREEN)) {
79  			if (event.getRawSlot() == InventoryHandler.INVENTORY_ADMIN_SLOT && view.getShop().getOwner().equals(player.getName())) {
80  				Conversation convo = CitiTrader.getConversationHandler().getAdminConversation().buildConversation(player);
81  				view.convo = convo;
82  				convo.begin();
83  				return;
84  			} else if (event.getRawSlot() == InventoryHandler.INVENTORY_ADMIN_SLOT && !view.getShop().getOwner().equals(player.getName())) {
85  				return;
86  			}
87  
88  			if (event.getCurrentItem() != null) {
89  				if (!event.getCursor().getType().equals(Material.AIR)) {
90  					player.getInventory().addItem(event.getCursor());
91  					player.setItemOnCursor(new ItemStack(Material.AIR));
92  				}
93  				view.buildItemView(event.getCurrentItem());
94  			} else if (event.getCursor().getData().getItemType().equals(Material.AIR)) {
95  			} else {
96  				event.setCancelled(false);
97  				ItemStack inhand = event.getCursor().clone();
98  				Conversation convo = CitiTrader.getConversationHandler().getAddSellItem().buildConversation(player);
99  				convo.getContext().setSessionData(ConversationHandler.CONVERSATION_SESSION_ITEM, inhand);
100 				convo.getContext().setSessionData(ConversationHandler.CONVERSATION_SESSION_SLOT, event.getRawSlot());
101 				view.convo = convo;
102 				convo.begin();
103 			}
104 		} else if (view.getStatus().equals(Status.SELL_SCREEN)) {
105 
106 			if (event.getRawSlot() == InventoryHandler.INVENTORY_BACK_ARROW_SLOT) {
107 				view.buildView();
108 				return;
109 			}
110 
111 			if (event.getRawSlot() == InventoryHandler.INVENTORY_ADMIN_SLOT && view.getShop().getOwner().equals(player.getName())) {
112 				Conversation convo = CitiTrader.getConversationHandler().getSetSellPrice().buildConversation(player);
113 				convo.getContext().setSessionData(ConversationHandler.CONVERSATION_SESSION_ITEM, event.getCurrentItem());
114 				view.convo = convo;
115 				convo.begin();
116 				return;
117 			} else if (event.getRawSlot() == InventoryHandler.INVENTORY_ADMIN_SLOT && !view.getShop().getOwner().equals(player.getName())) {
118 				return;
119 			}
120 
121 			if (event.getRawSlot() == InventoryHandler.INVENTORY_TAKE_ALL_SLOT && view.getShop().getOwner().equals(player.getName())) {
122 				// Put all items into inventory
123 				return;
124 			} else if (event.getRawSlot() == InventoryHandler.INVENTORY_TAKE_ALL_SLOT && !view.getShop().getOwner().equals(player.getName())) {
125 				return;
126 			}
127 
128 			if (event.getCurrentItem() != null) {
129 				PurchaseHandler.processPurchase(view.getShop(), (Player) event.getWhoClicked(), event.getCurrentItem());
130 				plugin.getServer().getScheduler().scheduleSyncDelayedTask(plugin, new Runnable() {
131 					@Override
132 					public void run() {
133 						view.buildItemView(event.getCurrentItem());
134 					}
135 				}, CitiTrader.BUKKIT_SCHEDULER_DELAY);
136 			}
137 		}

150 	@EventHandler
151 	public void onPlayerInteractEntityEvent(PlayerInteractEntityEvent event) {
152 		Entity entity = event.getRightClicked();
153 		Player player = event.getPlayer();
154 
155 		// Check if player can even access stores
156 		if (!player.hasPermission(Permissions.USE_STORES)) {
157 			return;
158 		}
159 
160 		if (plugin.isCitizens()) {
161 			if (!CitizensAPI.getNPCRegistry().isNPC(entity)) {
162 				return;
163 			}
164 
165 			NPC npc = CitizensAPI.getNPCRegistry().getNPC(entity);
166 
167 			if (npc.hasTrait(TraderTrait.class)) {
168 				TraderTrait trait = npc.getTrait(TraderTrait.class);
169 
170 				Integer shopid = trait.getShopId();
171 
172 				if (shopid == ShopHandler.SHOP_NULL) {
173 					if (npc.getTrait(Owner.class).isOwnedBy(player)) {
174 						Conversation convo = CitiTrader.getConversationHandler().getCreateShop().buildConversation(player);
175 						convo.getContext().setSessionData(ConversationHandler.CONVERSATION_SESSION_NPC, npc);
176 						convo.begin();
177 						return;
178 					} else {
179 						player.sendMessage(CitiTrader.getResourceBundle().getString("general.noopen.unassigned"));
180 						return;
181 					}
182 				}
183 
184 				InventoryHandler handler = CitiTrader.getStoreHandler().getInventoryHandler();
185 				// Open Store
186 				if (handler.hasInventoryView(player)) {
187 					ShopInventoryView view = (ShopInventoryView) handler.getInventoryView(player);
188 					if (!view.getShop().getId().equals(shopid)) {
189 						handler.createBuyInventoryView(player, CitiTrader.getStoreHandler().getShop(shopid));
190 					}
191 				} else {
192 					handler.createBuyInventoryView(player, CitiTrader.getStoreHandler().getShop(shopid));
193 				}
194 
195 				handler.openInventory(player);
196 			}
197 		}

201 	 * Checks when a player interacts with a block or item (in hand) to see
202 	 * if it is a store.
203 	 *
204 	 * @param event
205 	 */
206 	@EventHandler
207 	public void onPlayerInteractEvent(PlayerInteractEvent event) {
208 		Player player = event.getPlayer();
209 
210 		// Check if player can even access stores
211 		if (!player.hasPermission(Permissions.USE_STORES)) {
212 			return;
213 		}
214 
215 		if (event.getAction().equals(Action.RIGHT_CLICK_BLOCK)) {
216 			Block block = event.getClickedBlock();
217 			if (block.hasMetadata("shop")) {
218 				return;
219 			}
220 		}
221 
222 		if (event.getAction().equals(Action.LEFT_CLICK_AIR) || event.getAction().equals(Action.LEFT_CLICK_BLOCK)) {
223 			ItemStack item = event.getItem();
224 
225 			// after here is all itemsinhand -- if no item in hand. return.
226 			if (item == null) {
227 				return;
228 			}
229 
230 			if (!item.hasItemMeta()) {
231 				return;
232 			}
233 
234 			switch (item.getItemMeta().getDisplayName()) {
235 				case "Store":
236 					// Open Store
237 					CitiTrader.getStoreHandler().getInventoryHandler().createBuyInventoryView(player, CitiTrader.getStoreHandler().getShop(1));
238 					CitiTrader.getStoreHandler().getInventoryHandler().openInventory(player);
239 					break;
240 				case "Create Shop":
241 					createShop(player, item, event.getClickedBlock());
242 					break;
243 				default:
244 			}
245 		}
246 	}

248 	public void createShop(Player player, ItemStack item, Block block) {
249 		if (!player.hasPermission(Permissions.CREATE_STORES)) {
250 			player.sendMessage(CitiTrader.getResourceBundle().getString("permission.create.deny"));
251 			return;
252 		}
253 		if (item.getItemMeta().hasLore()) {
254 			ItemMeta meta = item.getItemMeta();
255 			Integer uses = Integer.valueOf(meta.getLore().get(2).split(" ")[0]) - 1;
256 			List<String> metas = meta.getLore();
257 			metas.set(2, String.valueOf(uses) + " uses left.");
258 			meta.setLore(metas);
259 			item.setItemMeta(meta);
260 		}

112 		Player player = event.getPlayer();
113 
114 		// Check if player can even access stores
115 		if (!player.hasPermission(Permissions.USE_STORES)) {
116 			return;
117 		}
118 
119 		if (event.getAction().equals(Action.RIGHT_CLICK_BLOCK)) {
120 			Block block = event.getClickedBlock();
121 			if (block.hasMetadata("shop")) {
122 				return;
123 			}
124 		}
125 
126 		if (event.getAction().equals(Action.LEFT_CLICK_AIR) || event.getAction().equals(Action.LEFT_CLICK_BLOCK)) {
127 			ItemStack item = event.getItem();
128 
129 			// after here is all itemsinhand -- if no item in hand. return.
130 			if (item == null) {
131 				return;
132 			}
133 
134 			if (!item.hasItemMeta()) {
135 				return;
136 			}
137 
138 			switch (item.getItemMeta().getDisplayName()) {
139 				case "Store":
140 					// Open Store
141 					inventoryHandler.createBuyInventoryView(player, plugin.getStoreHandler().getShop(1));
142 					inventoryHandler.openInventory(player);
143 					break;
144 				case "Create Shop":
145 					createShop(player, item, event.getClickedBlock());
146 					break;
147 			}
148 		}
149 	}

151 	public void createShop(Player player, ItemStack item, Block block) {
152 		if (!player.hasPermission(Permissions.CREATE_STORES)) {
153 			player.sendMessage("You do not have permission to open a shop.");
154 			return;
155 		}
156 		if (item.getItemMeta().hasLore()) {
157 			ItemMeta meta = item.getItemMeta();
158 			Integer uses = Integer.valueOf(meta.getLore().get(2).split(" ")[0]) - 1;
159 			List<String> metas = meta.getLore();
160 			metas.set(2, String.valueOf(uses) + " uses left.");
161 			meta.setLore(metas);

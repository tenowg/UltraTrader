13  /**
14   * 
15   * If there is anything left (no order is required) a reverse order will be
16   * done, and price will be reduced by anything that remains.
17   * 
18   * Order for secure transactions:
19   * 
20   * 1) figure price for all items in "Buy Box" - "Buy Stack"
21   * 2) Check if player has available funds
22   * 3) Withdraw funds - hold funds in temp variable *
23   * 4) Remove items (also check for availability) from traders inventory *
24   * 5) If difference, remove from players "Buy Box"
25   * 6) Attempt to put all items in players inventory
26   * 7) Any remaining items not put into inventory - figure price value
27   * 8) subtract remaining value from holding
28   * 9) refund funds to player account
29   * 10) deposit funds into trader account
30   * 11) return any remaining items to trader inventory.
31   */
32  public class PurchaseHandler {
33  	public static void processPurchase(Shop shop, Player player, ItemStack buyStack) {
34  		String buyStackId = shop.getItemId(buyStack);
35  		InventoryItem invItem = shop.getInventory().get(buyStackId);
36  		Double buyStackPriceEach = invItem.getSellprice();
37  		
38  		Double buyStackPrice = buyStackPriceEach * buyStack.getAmount();
39  		
40  		EconomyResponse heldFunds;
41  		if (CitiTrader.getEconomy().has(player.getName(), player.getWorld().getName(), buyStackPrice)) {
42  			heldFunds = CitiTrader.getEconomy().withdrawPlayer(player.getName(), player.getWorld().getName(), buyStackPrice);
43  		} else {
44  			player.sendMessage("You do not have the funds for this Transaction.");
45  			return;
46  		}
47  		
48  		if (heldFunds == null) {
49  			player.sendMessage("There was an error with the economy, please report. (NULL)");
50  			return;
51  		}
52  		
53  		if (!heldFunds.type.equals(ResponseType.SUCCESS)) {
54  			player.sendMessage("There was an error with the economy, please report.");
55  			return;
56  		}
57  		
58  		// Check for availability
59  		if (buyStack.getAmount() > invItem.getAmount()) {
60  			CitiTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), heldFunds.amount);
61  			// Some error checks for deposit
62  			player.sendMessage("There are not enough items to purchase. (Store is out of stock)");
63  			return;
64  		} else {
65  			invItem.setAmount(invItem.getAmount() - buyStack.getAmount());
66  		}
67  		
68  		ItemStack placedStack = invItem.getItemStack();
69  		placedStack.setAmount(buyStack.getAmount());
70  		HashMap<Integer, ItemStack> returnedItems = player.getInventory().addItem(placedStack);
71  		
72  		Double refund = 0D;
73  		if (!returnedItems.isEmpty()) {
74  			// calculate returned funds... readd to merchant
75  			Integer returned = 0;
76  			for (ItemStack returnedItem : returnedItems.values()) {
77  				returned = returned + returnedItem.getAmount();
78  			}
79  			
80  			refund = returned * buyStackPriceEach;
81  			
82  			CitiTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), refund);
83  			
84  			invItem.setAmount(invItem.getAmount() + returned);
85  		}
86  		
87  		Double traderDeposit = heldFunds.amount - refund;
88  		
89  		// Deposit into trader when interface is ready.
90  		player.sendMessage("Purchase Complete...");
91  		player.sendMessage("Trader recieved: " + traderDeposit);
92  	}

20  @WalletTypeName("shop")

22  public class ShopWallet extends Wallet {
23

27  	public ShopWallet(Shop shop) {
28  		super(shop);

32  	@Override
33  	public boolean hasFunds(BigDecimal amount) {
34  		BigDecimal balance = getBalance();
35  		if (balance.compareTo(amount) > 0) {
36  			return true;
37  		}
38  		return false;
39  	}
40  
41  	@Override
42  	public EconomyResponse removeFunds(BigDecimal amount) {
43  		BigDecimal balance = getBalance();
44  		if (hasFunds(amount)) {
45  			balance = balance.subtract(amount);
46  			double value = balance.doubleValue();
47  			shop.getWalletinfo().put("balance", new ConfigValue(value));
48  			return new EconomyResponse(amount.doubleValue(), getBalance().doubleValue(), ResponseType.SUCCESS, "");
49  		}
50  
51  		return new EconomyResponse(amount.doubleValue(), getBalance().doubleValue(), ResponseType.FAILURE, "Not enough funds");
52  	}
53  
54  	@Override
55  	public EconomyResponse addFunds(BigDecimal amount) {
56  		BigDecimal balance = getBalance();
57  		balance = balance.add(amount);
58  		double value = balance.doubleValue();
59  		shop.getWalletinfo().put("balance", new ConfigValue(value));
60  		return new EconomyResponse(amount.doubleValue(), getBalance().doubleValue(), ResponseType.SUCCESS, "");
61  	}
62  
63  	private BigDecimal getBalance() {
64  		//BigDecimal balance = BigDecimal.valueOf(config.getDouble("balance", 0D));
65  		ConfigValue<Double> get = shop.getWalletinfo().get("balance");
66  		if (get == null) {
67  			get = new ConfigValue<>(0D);
68  		}
69  		BigDecimal balance = BigDecimal.valueOf(get.getValue());
70  		return balance;
71  	}

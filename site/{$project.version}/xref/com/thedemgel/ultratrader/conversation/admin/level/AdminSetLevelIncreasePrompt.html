22  	private double price;
23  	private Shop shop;
24  	private int nextlevel;
25  	private ConversationPrefix prefix;
26  
27  	public AdminSetLevelIncreasePrompt() {
28  		prefix = new AdminConversationPrefix();
29  	}
30  
31  	@Override
32  	protected Prompt acceptValidatedInput(ConversationContext context, boolean input) {
33  		if (!input) {
34  			return new AdminSetLevelPrompt();
35  		}
36  
37  		// Take the money
38  		EconomyResponse resp = UltraTrader.getEconomy().withdrawPlayer(player.getName(), player.getWorld().getName(), price);
39  
40  		if (resp.type.equals(ResponseType.SUCCESS)) {
41  		// Increase the level
42  			shop.setLevel(nextlevel);
43  			player.sendRawMessage(prefix.getPrefix(context) + L.getFormatString("conversation.admin.level.raised", nextlevel));
44  		} else {
45  			player.sendRawMessage(prefix.getPrefix(context) + L.getString("conversation.admin.level.notenoughfunds"));
46  		}
47  		return new AdminSetLevelPrompt();
48  	}
49  
50  	@Override
51  	public String getPromptText(ConversationContext context) {
52  		ShopInventoryView view = (ShopInventoryView) context.getSessionData(ConversationHandler.CONVERSATION_SESSION_VIEW);
53  
54  		shop = view.getShop();
55  		player = (Player) context.getForWhom();
56  		nextlevel = shop.getLevel() + 1;
57  
58  		price = LimitHandler.getLevelCost(player, nextlevel);
59  
60  		return L.getFormatString("conversation.admin.level.proceed", price, nextlevel);
61  	}
